## 1. 项目概述

### 1.1 项目背景
本项目旨在开发一款仿照Microsoft OneNote的鸿蒙原生笔记应用，为用户提供跨平台的笔记管理解决方案。应用将充分利用鸿蒙系统的分布式能力和特性，提供流畅的笔记体验。

### 1.2 核心功能
- **多层级笔记本结构**: 支持创建多个笔记本，每个笔记本包含多个分区和页面
- **富文本编辑**: 支持文本格式化、图片插入、手写笔记、待办事项
- **标签管理**: 为笔记添加标签，便于分类和检索
- **全文搜索**: 支持关键词搜索和全文检索
- **云端同步**: 利用鸿蒙云服务实现数据同步
- **附件管理**: 支持图片、文档等多种附件类型
- **导出功能**: 支持导出为PDF、文本等格式

### 1.3 技术目标
- 使用鸿蒙SDK 5.0.4开发，确保系统兼容性
- 采用MVVM架构模式，实现代码分层
- 遵循鸿蒙开发规范，确保应用质量
- 实现良好的性能和用户体验
- 确保数据安全和隐私保护

#### 4.3.1 服务结构
每个服务类应遵循以下结构：

```typescript
/**
 * 笔记本服务
 * 提供笔记本相关的业务逻辑处理
 */
export class NotebookService {
  private static instance: NotebookService;
  private notebookDao: NotebookDao;
  
  // 私有构造函数，实现单例模式
  private constructor() {
    this.notebookDao = NotebookDao.getInstance();
  }
  
  /**
   * 获取服务实例
   */
  public static getInstance(): NotebookService {
    if (!NotebookService.instance) {
      NotebookService.instance = new NotebookService();
    }
    return NotebookService.instance;
  }
  
  /**
   * 创建新笔记本
   */
  public async createNotebook(title: string): Promise<Notebook> {
    // 参数验证
    if (!title || title.trim().length === 0) {
      throw new Error('笔记本标题不能为空');
    }
    
    // 创建数据模型
    const notebook = new Notebook(
      this.generateId(),
      title.trim(),
      '',
      this.getRandomColor(),
      'notebook'
    );
    
    // 保存到数据库
    await this.notebookDao.insert(notebook);
    
    return notebook;
  }
  
  /**
   * 获取所有笔记本
   */
  public async getAllNotebooks(): Promise<Notebook[]> {
    return await this.notebookDao.getAll();
  }
  
  /**
   * 更新笔记本
   */
  public async updateNotebook(notebook: Notebook): Promise<void> {
    if (!notebook.validate()) {
      throw new Error('笔记本数据无效');
    }
    
    notebook.modifiedTime = Date.now();
    await this.notebookDao.update(notebook);
  }
  
  /**
   * 删除笔记本
   */
  public async deleteNotebook(id: string): Promise<void> {
    // 检查笔记本是否存在
    const notebook = await this.notebookDao.getById(id);
    if (!notebook) {
      throw new Error('笔记本不存在');
    }
    
    // 删除笔记本及其所有页面
    await this.notebookDao.delete(id);
    // TODO: 删除关联的页面
  }
  
  /**
   * 生成唯一ID
   */
  private generateId(): string {
    return `notebook_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`;
  }
  
  /**
   * 获取随机颜色
   */
  private getRandomColor(): string {
    const colors = [
      '#0078D4', '#E81123', '#107C10', '#FFB900', 
      '#FF8C00', '#8764B8', '#00B7C3', '#0078D4'
    ];
    return colors[Math.floor(Math.random() * colors.length)];
  }
}
```

```typescript
    CREATE TABLE IF NOT EXISTS ${DatabaseConstants.TABLE_NOTEBOOKS} (
      ${DatabaseConstants.FIELD_ID} TEXT PRIMARY KEY,
      ${DatabaseConstants.FIELD_TITLE} TEXT NOT NULL,
      ${DatabaseConstants.FIELD_DESCRIPTION} TEXT,
      ${DatabaseConstants.FIELD_COLOR} TEXT,
```

#### 4.4.2 数据访问对象（DAO）
```typescript
/**
 * 笔记本数据访问对象
 */
export class NotebookDao {
  private static instance: NotebookDao;
  private rdbStore: relationalStore.RdbStore | null = null;
  
  private constructor() {}
  
  public static getInstance(): NotebookDao {
    if (!NotebookDao.instance) {
      NotebookDao.instance = new NotebookDao();
    }
    return NotebookDao.instance;
  }
  
  /**
   * 初始化数据库
   */
  public async init(): Promise<void> {
    if (this.rdbStore) {
      return;
    }
    
    const config: relationalStore.StoreConfig = {
      name: DatabaseConstants.DATABASE_NAME,
      securityLevel: relationalStore.SecurityLevel.S1
    };
    
    this.rdbStore = await relationalStore.getRdbStore(getContext(), config);
    await this.createTables();
  }
  
  /**
   * 创建表
   */
  private async createTables(): Promise<void> {
    if (!this.rdbStore) {
      throw new Error('数据库未初始化');
    }
    
    await this.rdbStore.executeSql(DatabaseConstants.SQL_CREATE_NOTEBOOKS_TABLE);
    await this.rdbStore.executeSql(DatabaseConstants.SQL_CREATE_INDEX);
  }
  
  /**
   * 插入笔记本
   */
  public async insert(notebook: Notebook): Promise<number> {
    if (!this.rdbStore) {
      await this.init();
    }
    
    const valueBucket: relationalStore.ValuesBucket = {
      [DatabaseConstants.FIELD_ID]: notebook.id,
      [DatabaseConstants.FIELD_TITLE]: notebook.title,
      [DatabaseConstants.FIELD_DESCRIPTION]: notebook.description,
      [DatabaseConstants.FIELD_COLOR]: notebook.color,
      [DatabaseConstants.FIELD_ICON]: notebook.icon,
      [DatabaseConstants.FIELD_CREATED_TIME]: notebook.createdTime,
      [DatabaseConstants.FIELD_MODIFIED_TIME]: notebook.modifiedTime
    };
    
    return await this.rdbStore.insert(DatabaseConstants.TABLE_NOTEBOOKS, valueBucket);
  }
  
  /**
   * 更新笔记本
   */
  public async update(notebook: Notebook): Promise<number> {
    if (!this.rdbStore) {
      await this.init();
    }
    
    const valueBucket: relationalStore.ValuesBucket = {
      [DatabaseConstants.FIELD_TITLE]: notebook.title,
      [DatabaseConstants.FIELD_DESCRIPTION]: notebook.description,
      [DatabaseConstants.FIELD_COLOR]: notebook.color,
      [DatabaseConstants.FIELD_ICON]: notebook.icon,
      [DatabaseConstants.FIELD_MODIFIED_TIME]: notebook.modifiedTime
    };
    
    const predicates = new relationalStore.RdbPredicates(DatabaseConstants.TABLE_NOTEBOOKS);
    predicates.equalTo(DatabaseConstants.FIELD_ID, notebook.id);
    
    return await this.rdbStore.update(valueBucket, predicates);
  }
  
  /**
   * 删除笔记本
   */
  public async delete(id: string): Promise<number> {
    if (!this.rdbStore) {
      await this.init();
    }
    
    const predicates = new relationalStore.RdbPredicates(DatabaseConstants.TABLE_NOTEBOOKS);
    predicates.equalTo(DatabaseConstants.FIELD_ID, id);
    
    return await this.rdbStore.delete(predicates);
  }
  
  /**
   * 查询所有笔记本
   */
  public async getAll(): Promise<Notebook[]> {
    if (!this.rdbStore) {
      await this.init();
    }
    
    const predicates = new relationalStore.RdbPredicates(DatabaseConstants.TABLE_NOTEBOOKS);
    predicates.orderByDesc(DatabaseConstants.FIELD_MODIFIED_TIME);
    
    const resultSet = await this.rdbStore.query(predicates);
    const notebooks: Notebook[] = [];
    
    if (resultSet.rowCount > 0) {
      resultSet.goToFirstRow();
      do {
        notebooks.push(this.fromResultSet(resultSet));
      } while (resultSet.goToNextRow());
    }
    
    resultSet.close();
    return notebooks;
  }
  
  /**
   * 根据ID查询笔记本
   */
  public async getById(id: string): Promise<Notebook | null> {
    if (!this.rdbStore) {
      await this.init();
    }
    
    const predicates = new relationalStore.RdbPredicates(DatabaseConstants.TABLE_NOTEBOOKS);
    predicates.equalTo(DatabaseConstants.FIELD_ID, id);
    
    const resultSet = await this.rdbStore.query(predicates);
    
    if (resultSet.rowCount === 0) {
      resultSet.close();
      return null;
    }
    
    resultSet.goToFirstRow();
    const notebook = this.fromResultSet(resultSet);
    resultSet.close();
    
    return notebook;
  }
  
  /**
   * 从ResultSet转换为Notebook对象
   */
  private fromResultSet(resultSet: relationalStore.ResultSet): Notebook {
    return new Notebook(
      resultSet.getString(resultSet.getColumnIndex(DatabaseConstants.FIELD_ID)),
      resultSet.getString(resultSet.getColumnIndex(DatabaseConstants.FIELD_TITLE)),
      resultSet.getString(resultSet.getColumnIndex(DatabaseConstants.FIELD_DESCRIPTION)),
      resultSet.getString(resultSet.getColumnIndex(DatabaseConstants.FIELD_COLOR)),
      resultSet.getString(resultSet.getColumnIndex(DatabaseConstants.FIELD_ICON)),
      resultSet.getLong(resultSet.getColumnIndex(DatabaseConstants.FIELD_CREATED_TIME)),
      resultSet.getLong(resultSet.getColumnIndex(DatabaseConstants.FIELD_MODIFIED_TIME))
    );
  }
}
```

```typescript
// 使用事务避免锁定
public async transaction(): Promise<void> {
  if (!this.rdbStore) {
    await this.init();

  } catch (error) {
    await this.rdbStore.rollBack();
    throw error;
  }
}
```
